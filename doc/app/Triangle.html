def initialize(*side_lengths)
  begin
    @a, @b, @c = side_lengths.map{|i|i.to_f}
    @A = angle_A.to_degrees.round(3)
    @C = inverse_cosC.to_degrees.round(3)
    @B = (180 - (@A + @C)).round(3)
    @coordinates = [
      {x: 0, y: 0},
      {x: @a, y: 0}, 
      third_vertex
    ]
  rescue
    @possible = false
  end
end

def third_vertex
  x = ((@b**2)+(@a**2)-(@c**2))/(2*@a)
  y = Math.sqrt((@b**2)-(x**2))
  { x: x.round(3), y: y.round(3) }
end
